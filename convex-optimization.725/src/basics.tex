\begin{headered_note}[Proof that Disc/Sphere is Convex]
  \begin{align*}
    ||x||=\sqrt{x_1^2+x_2^2}\\
    ||x||\leq 1\text{, and }||y||\leq 1\text{, and }\alpha \in [0,1]\\
    \implies ||\alpha x+(1-\alpha)y||\leq \alpha ||x||+(1-\alpha)||y||\leq 1
  \end{align*}
\end{headered_note}
